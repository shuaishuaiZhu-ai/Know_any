### 1. 概述

FlagCX (Flag Cross-Chip Communication Library) 通过灵活的适配器(Adaptor)架构支持多种CCL后端,实现异构GPU集群的统一通信接口。本文档基于FlagCX源码分析,提供完整的后端对接指南。本文基于: FlagCX v0.7+ 。

### 1.1 支持的CCL后端

FlagCX已对接的10+种CCL后端:

| 后端 | 硬件平台 | 对应设备适配器 | 编译标志 | 成熟度 |
| --- | --- | --- | --- | --- |
| NCCL | NVIDIA GPU | CUDA | USE_NVIDIA=1 | 完整 |
| IXCCL | Iluvatar CoreX | IXCUDA | USE_ILUVATAR_COREX=1 | 完整 |
| CNCL | Cambricon MLU | MLU | USE_CAMBRICON=1 | 部分功能未实现 |
| MCCL | MetaX MACA | MACA | USE_METAX=1 | 完整 |
| XCCL | Kunlunxin XPU | Kunlun | USE_KUNLUNXIN=1 | 完整 |
| DUCCL | DU GPU | DUCUDA | USE_DU=1 | 完整 |
| HCCL | Huawei Ascend | CANN | USE_ASCEND=1 | 部分功能未实现 |
| MUSACCL | Moore Threads MUSA | MUSA | USE_MUSA=1 | 完整 |
| RCCL | AMD GPU | HIP | USE_AMD=1 | 完整 |
| TCCL | TSMicro GPU | TSMicro | USE_TSM=1 | 完整 |

### 2. CCL适配器接口规范

### 2.1 适配器结构定义

CCL适配器通过`flagcxCCLAdaptor`结构体定义统一接口:

```

  // ===== 基础函数 =====
  flagcxResult_t (*getVersion)(int *version);

  const char *(*getErrorString)(flagcxResult_t result);

  flagcxResult_t (*reduce)(const void *sendbuff, void *recvbuff, size_t count,

                           int root, flagcxInnerComm_t comm,

  flagcxResult_t (*send)(const void *sendbuff, size_t count,

                         flagcxInnerComm_t comm, flagcxStream_t stream);
  flagcxResult_t (*recv)(void *recvbuff, size_t count,
                         flagcxDataType_t datatype, int peer,
```

### 3. NCCL后端对接示例

### 3.1 NCCL适配器完整实现

NCCL是最成熟的参考实现,展示了标准的对接模式:

```
<span></span>// File: flagcx/adaptor/ccl/nccl_adaptor.cc
#include "nvidia_adaptor.h"

#ifdef USE_NVIDIA_ADAPTOR

// ===== 基础函数 =====
flagcxResult_t ncclAdaptorGetVersion(int *version) {
  return (flagcxResult_t)ncclGetVersion(version);
}

flagcxResult_t ncclAdaptorGetUniqueId(flagcxUniqueId_t *uniqueId) {
  if (*uniqueId == NULL) {
    flagcxCalloc(uniqueId, 1);
  }
  return (flagcxResult_t)ncclGetUniqueId((ncclUniqueId *)(*uniqueId));
}

const char *ncclAdaptorGetErrorString(flagcxResult_t result) {
  return ncclGetErrorString((ncclResult_t)result);
}

// ===== 通信器管理 =====
flagcxResult_t ncclAdaptorCommInitRank(flagcxInnerComm_t *comm, int nranks,
                                       flagcxUniqueId_t commId, int rank,
                                       bootstrapState *bootstrap) {
  if (*comm == NULL) {
    flagcxCalloc(comm, 1);
  }
  return (flagcxResult_t)ncclCommInitRank(&(*comm)->base, nranks,
                                          *(ncclUniqueId *)commId, rank);
}

flagcxResult_t ncclAdaptorCommDestroy(flagcxInnerComm_t comm) {
  return (flagcxResult_t)ncclCommDestroy(comm->base);
}

// ===== 集合通信操作 =====
flagcxResult_t ncclAdaptorAllReduce(const void *sendbuff, void *recvbuff,
                                    size_t count, flagcxDataType_t datatype,
                                    flagcxRedOp_t op, flagcxInnerComm_t comm,
                                    flagcxStream_t stream) {
  return (flagcxResult_t)ncclAllReduce(
      sendbuff, recvbuff, count, (ncclDataType_t)datatype, (ncclRedOp_t)op,
      comm->base, stream->base);
}

// Gather需要通过Send/Recv实现(NCCL不直接支持)
flagcxResult_t ncclAdaptorGather(const void *sendbuff, void *recvbuff,
                                 size_t count, flagcxDataType_t datatype,
                                 int root, flagcxInnerComm_t comm,
                                 flagcxStream_t stream) {
  int rank, nranks;
  ncclResult_t res = ncclSuccess;
  res = ncclCommUserRank(comm->base, &rank);
  res = ncclCommCount(comm->base, &nranks);

  size_t size = count * getFlagcxDataTypeSize(datatype);
  char *buffer = static_cast<char *>(recvbuff);

  res = ncclGroupStart();
  if (rank == root) {
    for (int r = 0; r < nranks; r++) {
      res = ncclRecv(static_cast<void *>(buffer + r * size), size, ncclChar, r,
                     comm->base, stream->base);
    }
  }
  res = ncclSend(sendbuff, size, ncclChar, root, comm->base, stream->base);
  res = ncclGroupEnd();

  return (flagcxResult_t)res;
}

// ===== 适配器注册 =====
struct flagcxCCLAdaptor ncclAdaptor = {
    "NCCL",
    // Basic functions
    ncclAdaptorGetVersion, ncclAdaptorGetUniqueId, ncclAdaptorGetErrorString,
    ncclAdaptorGetLastError,
    // Communicator functions
    ncclAdaptorCommInitRank, ncclAdaptorCommFinalize, ncclAdaptorCommDestroy,
    ncclAdaptorCommAbort, ncclAdaptorCommResume, ncclAdaptorCommSuspend,
    ncclAdaptorCommCount, ncclAdaptorCommCuDevice, ncclAdaptorCommUserRank,
    ncclAdaptorCommGetAsyncError, ncclAdaptorMemAlloc, ncclAdaptorMemFree,
    ncclAdaptorCommRegister, ncclAdaptorCommDeregister,
    // Communication functions
    ncclAdaptorReduce, ncclAdaptorGather, ncclAdaptorScatter,
    ncclAdaptorBroadcast, ncclAdaptorAllReduce, ncclAdaptorReduceScatter,
    ncclAdaptorAllGather, ncclAdaptorAlltoAll, ncclAdaptorAlltoAllv,
    ncclAdaptorSend, ncclAdaptorRecv,
    // Group semantics
    ncclAdaptorGroupStart, ncclAdaptorGroupEnd};

#endif // USE_NVIDIA_ADAPTOR

```

```
<span></span>// 通信器初始化(CNCL特殊接口)

                                       flagcxUniqueId_t commId, int rank,
```

### 4.3 处理API差异

不同后端的API可能存在差异,需要适配:

| FlagCX接口 | NCCL | CNCL | HCCL | 处理方式 |
| --- | --- | --- | --- | --- |
| commInitRank | ncclCommInitRank | cnclInitComms | HcclCommInitRootInfo | 参数适配 |
| getUniqueId | ncclGetUniqueId | cnclGetCliqueId | HcclGetRootInfo | 直接映射 |
| memAlloc | ncclMemAlloc | 不支持 | 不支持 | 返回flagcxNotSupported |
| commFinalize | ncclCommFinalize | 不支持 | 不支持 | 返回错误码 |
| gather | 通过Send/Recv实现 | 通过Send/Recv实现 | 通过HcclBatchSendRecv实现 | 组合实现 |

### 5.2 特殊操作实现

HCCL的Gather操作使用批量Send/Recv:

```
<span></span>flagcxResult_t hcclAdaptorGather(const void *sendbuff, void *recvbuff,
                                 size_t count, flagcxDataType_t datatype,
                                 int root, flagcxInnerComm_t comm,
                                 flagcxStream_t stream) {
  uint32_t rank, nranks;
  HcclResult res = HCCL_SUCCESS;
  res = HcclGetRankSize(comm->base, &nranks);
  res = HcclGetRankId(comm->base, &rank);

  size_t size = count * getFlagcxDataTypeSize(datatype);
  char *buffer = static_cast<char *>(recvbuff);

  // 构建批量发送接收项
  std::vector<HcclSendRecvItem> sendRecvInfo;

  if (rank == root) {
    // Root接收所有rank的数据
    for (uint32_t r = 0; r < nranks; r++) {
      sendRecvInfo.emplace_back(HcclSendRecvItem{
          HcclSendRecvType::HCCL_RECV,
          static_cast<void *>(buffer + r * size),
          size,
          HcclDataType::HCCL_DATA_TYPE_INT8,
          r
      });
    }
  }

  // 所有rank发送数据到root
  void *sendbuffptr = (void *)sendbuff;
  sendRecvInfo.emplace_back(HcclSendRecvItem{
      HcclSendRecvType::HCCL_SEND,
      sendbuffptr,
      size,
      HcclDataType::HCCL_DATA_TYPE_INT8,
      (uint32_t)root
  });

  // 执行批量Send/Recv
  uint32_t itemNum = sendRecvInfo.size();
  HcclBatchSendRecv(sendRecvInfo.data(), itemNum, comm->base, stream->base);

  return (flagcxResult_t)h2f_ret_map[res];
}

```

### 6.2 第二步: 实现适配器

创建 `flagcx/adaptor/ccl/yourccl_adaptor.cc`:

```
<span></span>#include "yourvendor_adaptor.h"

    {flagcxSum, YOUR_SUM},
    {flagcxProd, YOUR_PROD},
    {flagcxMax, YOUR_MAX},
    {flagcxMin, YOUR_MIN}

std::map<YourCCLResult, flagcxResult_t> result_map = {
    {YOUR_SUCCESS, flagcxSuccess},
    {YOUR_ERROR, flagcxInternalError},
    // ... 完整映射

// ===== Step 2: 实现基础函数 =====

  YourCCLResult res = yourCCLGetVersion(version);

flagcxResult_t yourCCLAdaptorGetUniqueId(flagcxUniqueId_t *uniqueId) {
  if (*uniqueId == NULL) {
    flagcxCalloc(uniqueId, 1);
  }
  YourCCLResult res = yourCCLGetUniqueId((YourCCLUniqueId *)(*uniqueId));
  return result_map[res];
}

const char *yourCCLAdaptorGetErrorString(flagcxResult_t result) {
  // 将FlagCX错误码转换为YourCCL错误码
  YourCCLResult yourResult = YOUR_SUCCESS;
  for (auto &pair : result_map) {
    if (pair.second == result) {
      yourResult = pair.first;
      break;
    }

  return yourCCLGetErrorString(yourResult);

// ===== Step 3: 实现通信器管理 =====
flagcxResult_t yourCCLAdaptorCommInitRank(flagcxInnerComm_t *comm, int nranks,
                                           flagcxUniqueId_t commId, int rank,
                                           bootstrapState *bootstrap) {
  if (*comm == NULL) {
```

```
  const char *buffer_in = static_cast<const char *>(sendbuff);
  char *buffer_out = static_cast<char *>(recvbuff);

  // 使用组语义批量提交所有Send/Recv
  yourCCLGroupStart();

  for (int r = 0; r < nranks; r++) {
    if (sendcounts[r] > 0) {
      yourCCLSend(buffer_in + sdispls[r] * size, sendcounts[r],
                  datatype_map[datatype], r, comm->base, stream->base);
    }
    if (recvcounts[r] > 0) {
      yourCCLRecv(buffer_out + rdispls[r] * size, recvcounts[r],
                  datatype_map[datatype], r, comm->base, stream->base);
    }
  }

  yourCCLGroupEnd();

  return flagcxSuccess;
}

```

```
class YourCCLErrorTracker {
public:
  static void pushError(const char *func, YourCCLResult res) {
    ErrorInfo info = {func, res, std::chrono::system_clock::now()};
    error_stack.push_back(info);

    // 限制栈大小
    if (error_stack.size() > MAX_ERROR_STACK_SIZE) {

    }
  }

private:

    const char *func;
    YourCCLResult result;
    std::chrono::system_clock::time_point timestamp;
  };

  static constexpr int MAX_ERROR_STACK_SIZE = 100;
};
```
